# Then run once: playwright install chromium

beautifulsoup4

# requests — lightweight HTTP for the competition list (bi-weekly POST endpoint)
requests

# xlwings — reads and writes RTSF_2026_Worksheet.xlsx while it remains open in Excel
xlwings
